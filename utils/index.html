# node遍历文件夹目录树

const fs = require('fs');
/**
 * 递归遍历目录树

// const directoryTree = getDirectoryTreeSync(__dirname + '/code');

/**
 * 同步递归目录树
 * @param {*} rootPath 根目录
 */
function getDirectoryTreeSync(rootPath) {
    const info = fs.statSync(rootPath)
    if (info.isFile()) {
        //如果是文件就读取文件,并返回字符串
        nameArr = rootPath.split('/')
        return {
            name: nameArr[nameArr.length - 1],
            path: rootPath,
            size: info.size,
            mtime: info.mtime
        }
    } else {
        const childrenArr = [];
        const childrenInfo = fs.readdirSync(rootPath);
        for (let i = 0; i < childrenInfo.length; i++) {
            childrenPath = rootPath + '/' + childrenInfo[i];
            childrenArr.push(getDirectoryTreeSync(childrenPath));
        }
        return {
            name: rootPath.split('/')[rootPath.split('/').length - 1],
            children: childrenArr
        }
    }
}


/**
 * 异步递归目录树
 * @param {*} rootPath 根目录
 * @param {*} callback 回调函数，参数1:err,参数2：data,返回目录树
 */
function getDirectoryTree(rootPath, callback) {
    fs.stat(rootPath, (err, info) => {
        if (err) {
            callback(err);
        } else if (info.isFile()) {
            //如果是文件就读取文件,并返回字符串
            nameArr = rootPath.split('/')
            callback(null, {
                name: nameArr[nameArr.length - 1],
                path: rootPath,
                size: info.size,
                mtime: info.mtime
            })
        } else {
            nameArr = rootPath.split('/')
            fs.readdir(rootPath, (err, childrenInfo) => {
                if (err) {
                    callback(err);
                }
                let childrenArr = [];
                let childLength = 0;
                for (let i = 0; i < childrenInfo.length; i++) {
                    childrenPath = rootPath + '/' + childrenInfo[i];
                    getDirectoryTree(childrenPath, function (err, data) {
                        childLength++;
                        childrenArr.push(data);
                        if (childLength == childrenInfo.length) {
                            callback(null, {
                                name: rootPath.split('/')[rootPath.split('/').length - 1],
                                children: childrenArr
                            });
                        }
                    })
                }

            })
        }
    })
}

# 原生animate封装(减速，匀速)

//目标减速动画函数，参数1：动画元素，参数2：目标对象，参数3，执行完动画要执行的函数，可以是回调函数
function animate(element, styleObj, callBack) {
    //开始前先清除动画，避免累加BUG
    clearInterval(element.animate);
    //创建动画
    element.animate = setInterval(function () {
        //假设法，标记
        var flag = true;
        //遍历对象
        for (var attr in styleObj) {
            //判断attr是不是zIndex属性,如果是就直接交换
            if (attr == 'zIndex') {
                var target = styleObj[attr];
                element.style[attr] = target;
                //判断属性值是不是opactiy属性，如果是则保留2位小数，不用+px
            } else if (attr == 'opacity') {
                //获取元素当前的attr值
                var current = 100 * parseFloat(window.getComputedStyle(element)[attr]);
                //获取目标对象值
                var target = 100 * styleObj[attr];
                //设置步数，目标值与当前值之差，减速效果
                var step = (target - current) / 10;
                //判断如果目标值大于当前值则向上取整，否则向下取整
                step = current < target ? Math.ceil(step) : Math.floor(step);
                //每次执行都加上步数
                current += step;
                //赋值给元素
                element.style[attr] = current / 100;
                //判断如果当前值未达到目标值则标记变为false
                if (target != current) {
                    flag = false;
                }
                //其他情况
            } else {
                //获取元素当前的attr值
                var current = parseFloat(window.getComputedStyle(element)[attr]);
                //获取目标对象值
                var target = styleObj[attr];
                //设置步数，目标值与当前值之差，减速效果
                var step = (target - current) / 10;
                //判断如果目标值大于当前值则向上取整，否则向下取整
                step = current < target ? Math.ceil(step) : Math.floor(step);
                //每次执行都加上步数
                current += step;
                //赋值给元素
                element.style[attr] = current + 'px';
                //判断如果当前值未达到目标值则标记变为false
                if (target != current) {
                    flag = false;
                }
            }
        }
        //判断所有的都到达目标值，则停止执行
        if (flag) {
            clearInterval(element.animate)
            //判断如果callBack是函数，则执行
            if (typeof callBack == 'function') {
                callBack()
            }
        }
    }, 20)
}

//目标匀速动画函数，参数1：动画元素，参数2：目标对象，参数3：每次执行的步数，参数4，执行完动画要执行的函数，可以是回调函数
function animateBalance(element, styleObj, step, callBack) {
    //开始前先清除动画，避免累加BUG
    clearInterval(element.animate);
    //创建动画
    element.animate = setInterval(function () {
        //假设法，标记
        var flag = true;
        //遍历对象
        for (var attr in styleObj) {
            //获取元素当前的attr值
            var current = parseFloat(window.getComputedStyle(element)[attr]);
            //获取目标对象值
            var target = styleObj[attr];
            //设置步数
            step = step || 40;
            //每次执行都加上或减去步数
            if (current < target) {
                current += step;
            } else {
                current -= step;
            }
            //赋值给元素
            element.style[attr] = Math.abs(target - current) < step ? (target + 'px') : (current + 'px');
            //判断如果当前值未达到目标值则标记变为false
            if (target != current) {
                flag = false;
            }
        }
        //判断所有的都到达目标值，则停止执行
        if (flag) {
            clearInterval(element.animate)
            //判断如果callBack是函数，则执行
            if (typeof callBack == 'function') {
                callBack()
            }
        }
    }, 20)
}

# 拓展（IE10）

window.requestAnimationFrame(callback)

定义：告诉浏览器您希望执行动画并请求浏览器在下一次重绘之前调用指定的函数来更新动画。该方法使用一个回调函数作为参数，这个回调函数会在浏览器重绘之前调用。 配合translate用更好

频率是1000/60*毫秒

callback

一个指定函数的参数，该函数在下次重新绘制动画时调用。这个回调函数只有一个传参，DOMHighResTimeStamp，指示requestAnimationFrame() 开始触发回调函数的当前时间（performance.now() 返回的时间）。

返回值：一个 long 整数，请求 ID ，是回调列表中唯一的标识。是个非零值，没别的意义。你可以传这个值给 window.cancelAnimationFrame() 以取消回调函数。

# WebKit浏览器或移动端的页面

在WebKit浏览器或移动端（绝大部分是WebKit内核的浏览器）的页面实现比较简单，可以直接使用WebKit的CSS扩展属性(WebKit是私有属性)-webkit-line-clamp ；注意：这是一个 不规范的属性，它没有出现在 CSS 规范草案中。

常见结合属性：

  1. display: -webkit-box; 必须结合的属性 ，将对象作为弹性伸缩盒子模型显示 。
  2. -webkit-box-orient 必须结合的属性 ，设置或检索伸缩盒对象的子元素的排列方式 。
  3. text-overflow: ellipsis;，可以用来多行文本的情况下，用省略号“…”隐藏超出范围的文本 。查看 text-overflow 属性文档：

# 跨浏览器兼容的方案

p {
    position:relative;
    line-height:1.4em;
    /* 3 times the line-height to show 3 lines */
    height:4.2em;
    overflow:hidden;
}
p::after {
    content:"...";
    font-weight:bold;
    position:absolute;
    bottom:0;
    right:0;
    padding:0 20px 1px 45px;
    background:url(http://newimg88.b0.upaiyun.com/newimg88/2014/09/ellipsis_bg.png) repeat-y;
}
  1. height高度最好是line-height的3倍；
  2. 结束的省略好用了半透明的png做了减淡的效果，或者设置背景颜色；
  3. IE6-7不显示content内容，所以要兼容IE6-7可以是在内容中加入一个标签，比如用<span class="line-clamp">...</span>去模拟；
  4. 要支持IE8，需要将::after替换成:after；
